                ============ Database relationship==============
Role
 └── User
      ├── Course (instructor)
      ├── Enrollment ── Course
      ├── Progress ── Lesson
      └── Submission ── Quiz

Course
 ├── Section
 │    └── Lesson
 ├── Enrollment
 └── Quiz
      ├── Question
      │     └── Answer
      └── Submission
            └── SubmissionAnswer


POST   /auth/signup            --> đăng ký
POST   /auth/signin            --> đăng nhập
POST   /auth/signout           --> đăng xuất
POST   /auth/refresh           --> refresh access token
GET    /auth/me                --> lấy info user hiện tại
PATCH  /auth/change-password   --> đổi mật khẩu

                    ===== user ======

GET    /users                  --> list users (admin)
GET    /users/:id              --> chi tiết user
PATCH  /users/:id              --> update profile
DELETE /users/:id              --> xóa user (admin)

GET    /users/:id/enrollments  --> danh sách khóa học đã đăng ký
GET    /users/:id/progress     --> tiến độ học
GET    /users/:id/submissions  --> lịch sử làm quiz

                    ===== courses ======

GET    /courses                --> list courses (filter, search, pagination)
GET    /courses/:id            --> chi tiết course
POST   /courses                --> tạo course (instructor)
PATCH  /courses/:id            --> update course
DELETE /courses/:id            --> xóa course
PATCH  /courses/:id/publish    --> publish/unpublish


                    ===== sections ======

GET    /courses/:id/sections
POST   /courses/:id/sections
PATCH  /courses/:courseId/sections/:sectionId
DELETE /courses/:courseId/sections/:sectionId
                    ======lessons ===========

GET    /courses/:courseId/sections/:sectionId/lessons
POST   /courses/:courseId/sections/:sectionId/lessons
PATCH  /courses/:courseId/sections/:sectionId/lessons/:lessonId
DELETE /courses/:courseId/sections/:sectionId/lessons/:lessonId

                    ===== enrollments =======

POST   /enrollments                        # student tự enroll
GET    /enrollments/my                     # danh sách course đã enroll
GET    /enrollments/:id
PATCH  /enrollments/:id/status             # cancel
GET    /courses/:id/enrollments            # instructor xem ai đã enroll

                    ======progress =========

GET    /progress/my?course_id=             # student xem progress theo course
POST   /progress/lessons/:lessonId/complete
DELETE /progress/lessons/:lessonId/complete


                    ======= quizzes =======
GET    /courses/:courseId/quizzes
GET    /courses/:courseId/quizzes/:quizId
POST   /courses/:courseId/quizzes          # instructor
PATCH  /courses/:courseId/quizzes/:quizId  # instructor
DELETE /courses/:courseId/quizzes/:quizId  # instructor

                    ======= questions =======

GET    /quizzes/:quizId/questions
POST   /quizzes/:quizId/questions
PATCH  /quizzes/:quizId/questions/:questionId
DELETE /quizzes/:quizId/questions/:questionId

                    ====== answers ========
POST   /questions/:questionId/answers
PATCH  /questions/:questionId/answers/:answerId
DELETE /questions/:questionId/answers/:answerId

                    ====== submissions ========

POST   /quizzes/:quizId/submissions        # student nộp bài
GET    /quizzes/:quizId/submissions/my     # student xem bài của mình
GET    /quizzes/:quizId/submissions        # instructor xem tất cả
GET    /submissions/:id                    # chi tiết bài nộp + submission_answers

                    ======= submissions_answers =======
